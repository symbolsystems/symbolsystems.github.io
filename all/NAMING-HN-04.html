Hungarian notation for byte types [NAMING-HN-04-3]

DESCRIPTION 

For unsigned char variables and parameters use "by" or "y" prefix.
For char or signed char variables and parameters use "c" prefix.
SINCE 

v7.0
BENEFITS 

Improves readability and maintainability.
EXAMPLE 

char g_cmyVar;                                  // Violation
char g_MyVariable;                              // Violation
static const char g_skMyVariable = 'a';         // Violation
class A {
    char m_MyVariable;                          // Violation
    static const char m_skMyVariable = 'a';     // Violation
    void foo( char MyParam ) {                  // Violation
        char MyVariable;                        // Violation
        static const char skMyVariable = 'a';   // Violation
    }
};

unsigned char g_bymyVar2;                                // Violation
unsigned char g_MyVariable2;                             // Violation
static const unsigned char g_skMyVariable2 = 0;          // Violation
class B {
    unsigned char m_MyVariable;                          // Violation
    static const unsigned char m_skMyVariable = 0;       // Violation
    void foo( unsigned char MyParam ) {                  // Violation
        unsigned char MyVariable;                        // Violation
        static const unsigned char skMyVariable = 0;     // Violation
    }
};
REPAIR 

char g_cMyVar;                                  // OK
char g_cMyVariable;                             // OK
static const char g_skcMyVariable = 'a';        // OK
class A {
    char m_cMyVariable;                         // OK
    static const char m_skcMyVariable = 'a';    // OK
    void foo( char cMyParam ) {                 // OK
        char cMyVariable;                       // OK
        static const char skcMyVariable = 'a';  // OK
    }
};

unsigned char g_byMyVar2;                                 // OK
unsigned char g_byMyVariable2;                            // OK
static const unsigned char g_skbyMyVariable2 = 0;         // OK
class B {
    unsigned char m_byMyVariable;                         // OK
    static const unsigned char m_skbyMyVariable = 0;      // OK
    void foo( unsigned char byMyParam ) {                 // OK
        unsigned char byMyVariable;                       // OK
        static const unsigned char skbyMyVariable = 0;    // OK
    }
};
