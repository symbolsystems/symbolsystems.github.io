Hungarian notation for long int pointer, array, or reference types [NAMING-HN-28-3]

DESCRIPTION 

For long int pointer, array, or reference 
variables and parameters use "li" prefix.
SINCE 

v7.0
BENEFITS 

Improves readability and maintainability.
EXAMPLE 

long int* g_plimyVar;                                 // Violation
long int* g_pMyVariable;                              // Violation
static const long int* g_spkMyVariable;               // Violation
class B {
    long int* m_pMyVariable;                          // Violation
    static const long int* m_spkMyVariable;           // Violation
    void foo( long int* pMyParam ) {                  // Violation
        long int* pMyVariable;                        // Violation
        static const long int* spkMyVariable;         // Violation
    }
};
REPAIR 

long int* g_pliMyVar;                                 // OK
long int* g_pliMyVariable;                            // OK
static const long int* g_spkliMyVariable;             // OK
class A {
    long int* m_pliMyVariable;                        // OK
    static const long int* m_spkliMyVariable;         // OK
    void foo( long int* pliMyParam ) {                // OK
        long int* pliMyVariable;                      // OK
        static const long int* spkliMyVariable;       // OK
    }
};
