C++ exceptions shall not be used [JSF-208-2]

DESCRIPTION 

Do not use C++ exceptions.
SINCE 

v7.1
BENEFITS 

Rule improves portability of the code.
EXAMPLE 

#include <iostream>

using namespace std;
int main()
{
    char *buf;
    try                                          // Violation
    {
        buf = new char[512];
        if( buf == 0 )
            throw "Memory allocation failure!";  // Violation
    }
    catch( char * str )                          // Violation
    {
        cout << "Exception raised: " << str << '\n';
    }
    // ...
    return 0;
}
REPAIR 

#include <iostream>

using namespace std;
int main()                                   // OK
{
	char *buf;
	buf = new char[512];
	if( buf == 0 )
		cout<< "Memory allocation failure!";

	return 0;
}
