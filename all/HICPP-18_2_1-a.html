Use high_integrity::thread in place of std::thread [HICPP-18_2_1-a-3]

DESCRIPTION 

The std::thread classâ€™s destructor will call the std::terminate class if the
thread owned by the std::thread class is still joinable. The default behavior
can be provided by using a wrapper class.

For example:
//high_integrity.h
namespace high_integrity {
	enum ThreadExec: int32_t
	{
		DETACH,
		JOIN,
	};
	template<ThreadExec thread_exec>
	class thread {
	public:
		template<class F, class ... Args>
		thread(F&& f, Args &&... args )
		: m_thread ( std :: forward <F> (f), std :: forward <Args >( args )...)
		{}
		thread ( thread const &) = delete;
		thread ( thread &&) = default;
		~ thread ()
		{
			if( m_thread . joinable ())
			{
				join_or_detach ();
			}
		}
		inline void join () {m_thread . join ();}
	private :
		inline void join_or_detach ();
	private :
		std::thread m_thread;
	};
	template<> void thread<ThreadExec::DETACH>::join_or_detach() {
		m_thread.detach();
	}
	template<> void thread<ThreadExec::JOIN>::join_or_detach() {
		m_thread.join();
	}
}

This rule identifies cases where std::thread could call std::terminate.
SINCE 

v10.4
BENEFITS 

This rule makes it easier to identify and fix potential issues.
EXAMPLE 

#include <cstdint>
#include <thread>

void f(int32_t);
int32_t main() {
    int32_t i;

    std::thread t(f, i); // Violation - potentially calls std::terminate
    return 0;
}
REPAIR 

#include "high_integrity.h"

void f(int32_t);
int32_t main() {
    int32_t i;

    high_integrity::thread t(f, i); // OK - will not call std::terminate
    return 0;
}
