Hungarian notation for void pointer types [NAMING-HN-42-3]

DESCRIPTION 

For void pointer variables and parameters use "v" prefix.
SINCE 

v7.0
BENEFITS 

Improves readability and maintainability.
EXAMPLE 

void* g_pvmy;                                     // Violation
void* g_pMyVariable;                              // Violation
static const void* g_skpMyVariable;               // Violation
class A {
    void* m_pMyVariable;                          // Violation
    static const void* m_skpMyVariable;           // Violation
    void foo( void* pMyParam ) {                  // Violation
        void* pMyVariable;                        // Violation
        static const void* skpMyVariable;         // Violation
    }
};
REPAIR 

void* g_pvMyV;                                    // OK
void* g_pvMyVariable;                             // OK
static const void* g_skpvMyVariable;              // OK
class A {
    void* m_pvMyVariable;                         // OK
    static const void* m_skpvMyVariable;          // OK
    void foo( void* pvMyParam ) {                 // OK
        void* pvMyVariable;                       // OK
        static const void* skpvMyVariable;        // OK
    }
};
