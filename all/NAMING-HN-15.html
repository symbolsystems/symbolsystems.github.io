Hungarian notation for dword pointer, array, or reference types [NAMING-HN-15-3]

DESCRIPTION 

For unsigned long int pointer, array, or reference variables and parameters use
"dw" prefix. For signed long int or long int pointer, array, or reference
variables and parameters use "li" prefix.
SINCE 

v7.0
BENEFITS 

Improves readability and maintainability.
EXAMPLE 

long* g_plimyVar;                                   // Violation
long* g_pMyVariable;                                // Violation
static const long* g_spkMyVariable;                 // Violation
class A {
    long* m_pMyVariable;                            // Violation
    static const long* m_spkMyVariable;             // Violation
    void foo( long* pMyParam ) {                    // Violation
        long* pMyVariable;                          // Violation
        static const long* spkMyVariable;           // Violation
    }
};

unsigned long* g_pdwmyVar2;                         // Violation
unsigned long* g_pMyVariable2;                      // Violation
static const unsigned long* g_spkMyVariable2;       // Violation
class B {
    unsigned long* m_pMyVariable;                   // Violation
    static const unsigned long* m_spkMyVariable;    // Violation
    void foo( unsigned long* pMyParam ) {           // Violation
        unsigned long* pMyVariable;                 // Violation
        static const unsigned long* spkMyVariable;  // Violation
    }
};
REPAIR 

long* g_pliMyVar;                                      // OK
long* g_pliMyVariable;                                 // OK
static const long* g_spkliMyVariable;                  // OK
class A {
    long* m_pliMyVariable;                             // OK
    static const long* m_spkliMyVariable;              // OK
    void foo( long* pliMyParam ) {                     // OK
        long* pliMyVariable;                           // OK
        static const long* spkliMyVariable;            // OK
    }
};

unsigned long* g_pdwMyVar2;                            // OK
