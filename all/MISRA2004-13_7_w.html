Boolean operations whose results are invariant shall not be permitted [MISRA2004-13_7_w-3]

DESCRIPTION 

"Boolean operations whose results are invariant shall not be permitted."
BENEFITS 

"If a Boolean operator yields a result that can be proven to be always "true"
or always "false", it is highly likely that there is a programming error."
EXAMPLE 

void myFunctionEnum(){
    
 	 enum ec1 {BLACK, WHITE, GREY, PINK = 15} col1;

 	 if(col1 < 0);   // Violation
	 if(col1 < 16);  // Violation
 	 if(col1 <= 15); // Violation
	 if(col1 <= 16); // Violation
 	 if(col1 > 15);  // Violation
	 if(col1 > 16);  // Violation
	 if(col1 >= 0);  // Violation
	 if(col1 >= 16); // Violation
}
REPAIR 

void myFunctionEnum(){
    
	enum ec1 {BLACK, WHITE, GREY, PINK = 15} col1;

	if(col1 < 1);   // OK
 	if(col1 < 15);  // OK
	if(col1 <= 0);  // OK
	if(col1 <= 1);  // OK
	if(col1 > 0);   // OK
	if(col1 > 1);   // OK
	if(col1 >= 1);  // OK
 	if(col1 >= 15); // OK
}
