A 'default' label, if it exists, shall appear as either the first or the last switch label of a switch statement [MISRAC2012-RULE_16_5-a-2]

DESCRIPTION 

If a switch statement has a 'default' label, then the label should be the first
or the last label.

See also: CODSTA-35
SINCE 

v9.5
BENEFITS 

This rule makes it easy to locate the default label within a switch statement.
EXAMPLE 

/* Examples of incorrect 'default' placements: */
void foo ( int i )
{
  switch ( i )
  {
    case 0:
    default:              /* Violation: 'default' not first nor last  */
      i = 8;
      break;
    case 1:
      i = 7;
      break;
  }
  switch ( i )
  {
    case 0:
      i = 8;
      break;
    default:              /* Violation: 'default' not first nor last  */
      i = 10;
      break;
    case 1:
      i = 7;
      break;
  }
  switch ( i )
  {
    case 0:
      i = 8;
      break;
    default:              /* Violation: 'default' not first nor last  */
    case 1:
      i = 7;
      break;
  }
}
REPAIR 

/* Examples of correct 'default' placements: */
void foo ( int i )
{
  switch ( i )
  {
    default:              /* OK: 'default' is first */
    case 0:
      i = 8;
      break;
    case 1:
      i = 7;
      break;
  }
  switch ( i )
  {
    default:              /* OK: 'default' is first */
      i = 10;
      break;
    case 0:
      i = 8;
      break;
    case 1:
      i = 7;
      break;
  }
  switch ( i )
  {
    case 0:
      i = 8;
      break;
    case 1:
    default:              /* OK: 'default' is last */
      i = 7;
      break;
  }
  switch ( i )
  {
    case 0:
      i = 8;
      break;
    case 1:
      i = 7;
      break;
    default:              /* OK: 'default' is last */
      i = 10;
      break;
  }
}

/* Lack of 'default' is not a violation of this rule: */
void bar ( int i )
{
  switch ( i )
  {
    case 0:
      i = 8;
      break;
    case 1:
      i = 7;
      break;
  }
}
