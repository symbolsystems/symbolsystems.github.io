Do not dereference pointer type expressions [PB-01-3]

DESCRIPTION 

Do not use dereferenced pointer type expressions.
BENEFITS 

Improves safety and readability of code.
EXAMPLE 

int* pFun();

void goo()
{
    int* ptr;
    int a, b, c;
    a = *ptr++;      // Violation
    b = *pFun();     // Violation
    c = *(ptr + 5);  // Violation
}
REPAIR 

int* pFun();

void goo()
{
    int* ptr;
    int* ptr2;
    int a, b, c;
    
    ptr++;
    a = *ptr;       // OK
    
    ptr2 = pFun();
    b = *ptr2;      // OK
    
    ptr2 = ptr + 5;
    c = *ptr2;      // OK
}
