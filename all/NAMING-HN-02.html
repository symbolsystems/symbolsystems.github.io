Hungarian notation for bool types [NAMING-HN-02-3]

DESCRIPTION 

For bool variables and parameters use "b" prefix.
SINCE 

v7.0
BENEFITS 

Improves readability and maintainability.
EXAMPLE 

bool g_bmyVar;                                  // Violation
bool g_MyVariable;                              // Violation
static const bool g_skMyVariable = true;        // Violation
class A {
    bool m_MyVariable;                          // Violation
    static const bool m_skMyVariable = true;    // Violation
    void foo( bool MyParam ) {                  // Violation
        bool MyVariable;                        // Violation
        static const bool skMyVariable = true;  // Violation
    }
};
REPAIR 

bool g_bMyVar;                                  // OK
bool g_bMyVariable;                             // OK
static const bool g_skbMyVariable = true;       // OK
class A {
    bool m_bMyVariable;                         // OK
    static const bool m_skbMyVariable = true;   // OK
    void foo( bool bMyParam ) {                 // OK
        bool bMyVariable;                       // OK
        static const bool skbMyVariable = true; // OK
    }
};
