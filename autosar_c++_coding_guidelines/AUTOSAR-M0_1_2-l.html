Boolean operations whose results are invariant shall not be permitted [AUTOSAR-M0_1_2-l-2]

DESCRIPTION 

"Boolean operations whose results are invariant shall not be permitted."
BENEFITS 

"If a Boolean operator yields a result that can be proven to be always "true"
or always "false", it is highly likely that there is a programming error."
EXAMPLE 

void myFunctionEnum(){
    
  	enum ec1 {BLACK, WHITE, GREY, PINK = 15} col1;

 	if(15 < col1);  // Violation
	if(16 < col1);  // Violation 
	if(0 <= col1);  // Violation
	if(16 <= col1); // Violation
	if(0 > col1);   // Violation
	if(16 > col1);  // Violation 
 	if(15 >= col1); // Violation
	if(16 >= col1); // Violation 
}
REPAIR 

void myFunctionEnum(){
    
	enum ec1 {BLACK, WHITE, GREY, PINK = 15} col1;

	if(0 < col1);   // OK
	if(1 < col1);   // OK
	if(1 <= col1);  // OK
 	if(15 <= col1); // OK
	if(1 > col1);   // OK
 	if(15 > col1);  // OK
	if(0 >= col1);  // OK
	if(1 >= col1);  // OK
}
